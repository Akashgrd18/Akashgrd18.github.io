Query to extract Customer M2 ID list

select ce.entity_id from customer_entity as ce LEFT JOIN customer_entity_int AS cei7 ON ce.entity_id = cei7.entity_id AND cei7.attribute_id = 194 LEFT JOIN customer_entity_varchar cev ON ce.entity_id = cev.entity_id and cev.attribute_id=767 where CONVERT_TZ(ce.created_at,'+00:00','+09:00') >= "2025-05-12 00:00:00" // PREVIOUS DAY DATE and CONVERT_TZ(ce.created_at,'+00:00','+09:00') < "2025-05-13 00:00:00" //CURRENT DAY DATE and ce.website_id =21 and ce.confirmation IS NULL;
